\section*{1.}
\begin{equation*}
	\begin{pmatrix}
		5 & 3 \\
		-2 & 4 \\
	\end{pmatrix}
	\begin{pmatrix}
		1 \\
		7 \\
	\end{pmatrix}
	=
	\begin{pmatrix}
		5 \cdot 1 + 3 \cdot 7 \\
		-2 \cdot 1 + 4 \cdot 7 \\
	\end{pmatrix}
	=
	\boxed{
		\begin{pmatrix}
			26 \\
			26 \\
		\end{pmatrix}
	}
\end{equation*}
